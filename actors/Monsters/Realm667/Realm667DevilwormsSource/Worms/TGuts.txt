actor TGut1
{
    PROJECTILE
    -NOGRAVITY
    -NOBLOCKMAP
    -NOTELEPORT
    +RANDOMIZE
    Radius 2
    Damage 0
    Speed 5
	Scale 0.4
    States
    {
    Spawn:
        GUT1 A 0 A_SetGravity (0.5)
        GUT1 A 0 ThrustThingZ (0,random(15,30), 0, 1)
        goto See 
    See:
        GUT1 ABCDEFGH 2
        loop
    Death:
        GUT1 A 1 A_SpawnItem("TFlesh1",0,0,0)
        Stop
    }
}
